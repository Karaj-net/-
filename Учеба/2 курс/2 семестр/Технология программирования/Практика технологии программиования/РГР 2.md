                List<object> values = new List<object>();

                {
                    columns.Add("Школа=?");
                    values.Add(TextBox8.Text);
                }

                    columns.Add("Математика=?");
                    values.Add(TextBox4.Text);

                    columns.Add("Изо=?");
                    values.Add(TextBox5.Text);

                if (!string.IsNullOrEmpty(TextBox6.Text))
                {
                    columns.Add("ИТ=?");
                    values.Add(TextBox6.Text);
                }

                    columns.Add("Технология=?");
                    values.Add(TextBox7.Text);

                    Label1.Text += "Нет данных для обновления </br>";

                OleDbCommand cmd = new OleDbCommand(str, cnn);
                foreach (var value in values)
                {
                    cmd.Parameters.AddWithValue("?", value);
                }
                cnn.Open();
                int k = cmd.ExecuteNonQuery();
                Label1.Text += String.Format("Изменена {0} запись(и) </br>", k);

        protected void Button5_Click(object sender, EventArgs e)
        {
            DataSet ds = new DataSet();
            try
            {
                cnn = new OleDbConnection();
                cnn.ConnectionString = System.Web.Configuration.WebConfigurationManager.ConnectionStrings["DBConnectionString"].ConnectionString;
                OleDbDataAdapter da = new OleDbDataAdapter("Select * from Таблица2", cnn);
                da.Fill(ds, "Stepuha");
                DataTable dt = ds.Tables["Stepuha"];
                int n = dt.Rows.Count;
                Contact st = new Contact();
                cnn.Open();
                for (int i = 0; i < n; i++)
                {
                    DataRow dr;
                    dr = dt.Rows[i];
                    st.Name = dr["Фамилия"].ToString();
                    st.Mat = Convert.ToInt32(dr["Математика"]);
                    st.Izo = Convert.ToInt32(dr["Изо"]);
                    st.IT = Convert.ToInt32(dr["ИТ"]);
                    st.Tex = Convert.ToInt32(dr["Технология"]);
                    st.School = Convert.ToInt32(dr["Школа"]);
                    double s = st.Stepuha();
                    dr["Стипендия"] = s;
                    string query = "Update Таблица2 set Стипендия = ? where Фамилия = ?";
                    OleDbCommand cmd = new OleDbCommand(query, cnn);
                    cmd.Parameters.AddWithValue("Стипендия", s);
                    cmd.Parameters.AddWithValue("Фамилия", st.Name);
                    cmd.ExecuteNonQuery();
                    cmd.Parameters.Clear();
                }
            }
            catch (Exception ex)
            {
                Label1.Text += String.Format("Ошибка {0} </br>", ex.Message);
            }
            finally
            { cnn.Close(); }

        }

    }
