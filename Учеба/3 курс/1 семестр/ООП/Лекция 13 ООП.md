# Делегаты при описании полей класса 
Рассмотрим использование делегатов при описании полей класса. Представим, что мы создаем определенный класс и в его коде мы генерируем сообщение и оно отправляется пользователю, мы знаем, что это сообщение. но мы не знаем на какие устройства нужно их отправлять, например, рассмотрим класс, работающий с расчетом в некотором банке, клиент может совершать какие-то действия. Пользователь может снимать и класть деньги при этом он должен получать соответствующие сообщение, при чем у каждого пользователя свои предпочтения куда его отправлять, поэтому разработчик долен создать универсальный класс, который подходит для всех пользователей, он может для этого использовать делегаты. При создании делегатов, этим делегатом в качестве значений могут присваиваться определенные методы

Для примера сначала мы рассмотрим класс без использования делегатов
В нашем классе предполагаются поля для фамилии суммы и пароля
Методы - пополнение счета и снятия, при каждом этом действие должно быть сообщение. Также опишем метод для организации сеанса
```C#
class Account1
{
	private string name;
	private int sum;
	private string password;
	
	public Account1(string name; int sum; string password)
	{
		this.name = name;
		this.sum = sum;
		this.password = password;
	}
	
	void Add(int s)
	{
		sum += s;
		string str = String.Format("Клиент {0}, пополнен: {1}", name, sum);
		Console.Writeline(str);
	}
	void Sub(int s)
	{
		if (s <= sum)
		{
			sum -= s;
			string str = String.Format("Клиент {0}, снято: {1}", name, sum);
			Console.Writeline(str);
		}
		else
		{
			string str = String.Format("Клиент {0}, средств недостаточно {1}", name, sum);
			Console.Writeline(str);
		}
	}
	public void Act()
	{
		Console.Writeline("Клиент {0}, введите пароль!", name);
		string pw = Console.Readline();
		string code = "0"
		if  (pw != password)
		{
			Console.Writeline("Неправильный пароль!");
		}
		else
		{
			do 
			{
				Console.Writeline("Введите код операции:\n
								   0 - выход;\n
								   1 -  пополнить\n
								   2 - снять");
				string code = Console.Readline();
				int s;
				if (code == 1 or == 2)
				{
					Console.Writeline("Введите сумму!")
					s = Int32.Parse(Console.Writeline())
				}
				if (code == s)
				{
					Add(s);
				}
				if (code == 2)
				{
					Sub(s);
				}
				while (code != 0);
			}
		}
	}
}

class Program
{
	static void Main()
	{
		Account1 = acc1 = new Account1("Иванов", 50, "123");
		acc1.Act();
		Account1 acc2 = new Acount1("Петров", 70, "456")
	}
}
```
