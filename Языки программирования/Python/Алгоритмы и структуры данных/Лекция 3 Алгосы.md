    if n == 1:  # –ë–∞–∑–æ–≤—ã–π —Å–ª—É—á–∞–π

# –•–≤–æ—Å—Ç–æ–≤–∞—è —Ä–µ–∫—É—Ä—Å–∏—è (–æ–ø—Ç–∏–º–∏–∑–∏—Ä—É–µ—Ç—Å—è –≤ —Ü–∏–∫–ª)
def tail_factorial(n, accumulator=1):

        return accumulator
    return tail_factorial(n - 1, n * accumulator)

### **2.2 –ë–∏–Ω–∞—Ä–Ω—ã–π –ø–æ–∏—Å–∫ (O(log n))**
**–¢—Ä–µ–±–æ–≤–∞–Ω–∏–µ**: –æ—Ç—Å–æ—Ä—Ç–∏—Ä–æ–≤–∞–Ω–Ω—ã–π –º–∞—Å—Å–∏–≤

# –ò—Ç–µ—Ä–∞—Ç–∏–≤–Ω–∞—è —Ä–µ–∞–ª–∏–∑–∞—Ü–∏—è
def binary_search(arr, target):
    low, high = 0, len(arr) - 1
    
    while low <= high:
        mid = (low + high) // 2
        if arr[mid] == target:
            return mid
        elif arr[mid] < target:
            low = mid + 1
        else:
            high = mid - 1
    return -1

# –ü—Ä–∏–º–µ—Ä –∏—Å–ø–æ–ª—å–∑–æ–≤–∞–Ω–∏—è

### **2.3 –†–µ–∫—É—Ä—Å–∏–≤–Ω—ã–π –±–∏–Ω–∞—Ä–Ω—ã–π –ø–æ–∏—Å–∫**
–≠–ª–µ–≥–∞–Ω—Ç–Ω–∞—è —Ä–µ–∞–ª–∏–∑–∞—Ü–∏—è —á–µ—Ä–µ–∑ —Ä–µ–∫—É—Ä—Å–∏—é:

```python
def recursive_binary_search(arr, target, low=0, high=None):
    if high is None:
        high = len(arr) - 1
    
    if low > high:
        return -1
    
    mid = (low + high) // 2
    if arr[mid] == target:
        return mid
    elif arr[mid] < target:
        return recursive_binary_search(arr, target, mid + 1, high)
    else:
        return recursive_binary_search(arr, target, low, mid - 1)

# –ü—Ä–∏–º–µ—Ä –∏—Å–ø–æ–ª—å–∑–æ–≤–∞–Ω–∏—è

```

def dfs(graph, node, visited=None):
    if visited is None:
        visited = set()
    visited.add(node)
    print(node, end=" ")

        if neighbor not in visited:

### **3.2 –†–µ–∫—É—Ä—Å–∏–≤–Ω—ã–π –ø–æ–∏—Å–∫ –≤ –±–∏–Ω–∞—Ä–Ω–æ–º –¥–µ—Ä–µ–≤–µ**
```python
class Node:
    def __init__(self, value):
        self.value = value
        self.left = None
        self.right = None

def search_in_tree(root, target):
    if root is None:
        return False
    if root.value == target:
        return True
    return search_in_tree(root.left, target) or search_in_tree(root.right, target)

# –ü—Ä–∏–º–µ—Ä –¥–µ—Ä–µ–≤–∞
root = Node(10)
root.left = Node(5)
root.right = Node(15)
root.left.right = Node(7)

print(search_in_tree(root, 7))  # True
print(search_in_tree(root, 12))  # False
```
