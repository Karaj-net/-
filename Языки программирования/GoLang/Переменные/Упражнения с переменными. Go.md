import "fmt"

func main() {

    fmt.Println("Мой год рождения:", birtYear)

    var (
        firstInicial   = "К"
        surnameInicial = "Ш"
    )
    fmt.Println("Инициалы =", firstInicial, surnameInicial)

    //математические вычисления
    var ageInDays int
    ageInDays = 365 * ageInYears
    fmt.Println("Мой возраст в днях:", ageInDays)
}
