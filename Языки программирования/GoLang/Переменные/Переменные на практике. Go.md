#Go 
- Несколько простых примеров работы с переменными
```go
package main // мы указываем, что у нас есть главный пакет и                   это позволяет испоьзовать функцию мэйн

import "fmt" //базовый пакет для Go. Пакет форматирования и                   используется для печати информации

func main() { //фкц main является первой фкц, которая                          запускается при запуске программы на Go

    var myName = "Jason"
    fmt.Println("my name is", myName)
     
    var name string = "Kathy"
    fmt.Println("name =", name)

    userName := "admin"
    fmt.Println("username =", userName)

    var sum int
    fmt.Println("the sum is", sum)

    a, b := 5, 7
    fmt.Println("число №1 =", a, "число №2 =", b)

    a, h := 4, 7
    fmt.Println("число №3 =", a, "число №4 =", h)

    sum := a + b + h
    fmt.Println("sum =", sum)
}
```
